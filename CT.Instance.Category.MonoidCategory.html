Require Import CT.Category.

The category for a given monoid. A monoid is exactly a category with one object.
Program Definition MonoidCategory {T} (M : Monoid) : Category :=
  {| ob := unit;
     mor := fun _ _ ⇒ T;
     comp := fun _ _ _ ⇒ mu M;
     id := fun a ⇒ one M;
  |}.
